git reset --hard HEAD~ # удаляем коммит и изменения

git remote show (показывает список удаленных репозиториев, привязанных к текущему)

ssh-keygen (генерация SSH-ключей)

git remote show origin (запрос к гиту)

git push -u origin main (создать связь)

git push (отправить на гитхаб)

git pull (забрать с гитхаб)
